-- Завдання 1.1: Вибрати всі стовпчики з таблиці products
SELECT * FROM products;

-- Завдання 1.2: Вибрати тільки стовпчики name, phone з таблиці shippers
SELECT name, phone FROM shippers;

-- Завдання 2: Знайти середнє, максимальне та мінімальне значення price таблиці products
SELECT 
    AVG(price) AS average_price,
    MAX(price) AS max_price,
    MIN(price) AS min_price
FROM products;

-- Завдання 3: Обрати унікальні значення category_id та price таблиці products. 
-- Сортування за спаданням ціни, вивід 10 рядків.
SELECT DISTINCT category_id, price 
FROM products
ORDER BY price DESC
LIMIT 10;

-- Завдання 4: Знайти кількість продуктів в цінових межах від 20 до 100
SELECT COUNT(*) AS product_count 
FROM products 
WHERE price BETWEEN 20 AND 100;

-- Завдання 5: Знайти кількість продуктів та середню ціну у кожного постачальника
SELECT 
    supplier_id, 
    COUNT(*) AS product_count, 
    AVG(price) AS average_price 
FROM products 
GROUP BY supplier_id;
